102 def _crc(self, data: bytes) -> int:
103 """
104 @brief Calculates the CRC-16/XMODEM (Polynomial 0x1021).
105 @param data The raw byte sequence to verify.
106 @return The calculated 16-bit CRC integer.
107 """
108 crc = 0x0000
109 poly = 0x1021
110 for b in data:
111 crc ^= (b << 8)
112 for _ in range(8):
113 if crc & 0x8000: crc = (crc << 1) ^ poly
114 else: crc <<= 1
115 crc &= 0xFFFF
116 return crc

132 def start_process(self) -> None:
133 """
134 @brief Main execution loop of the scanner thread.
135 @details Steps:
136 1. Connect.
137 2. Passive Sniffing (30s).
138 3. Device Enumeration.
139 4. Active Scanning (Smart Scan).
140 5. Cleanup.
141 """
142 self.running = True
143 if not self.connect():
144 self.running = False
145 return
146
147 # --- PHASE 1: SNIFFING ---
148 self._log(f"PHASE 1: Network Sniffing ({SNIFF_DURATION}s)...")
149 start_time = time.time()
150 buf = bytearray()
151
152 while time.time() - start_time < SNIFF_DURATION and self.running:
153 remaining = int(SNIFF_DURATION - (time.time() - start_time))
154 # 50% of the progress bar is allocated to sniffing
155 prog = ((SNIFF_DURATION - remaining) / SNIFF_DURATION) * 50
156 self._log(f"Listening... {remaining}s remaining", prog)
157
158 try:
159 if self.sock:
160 chunk = self.sock.recv(4096)
161 if chunk:
162 buf.extend(chunk)
163 # Process buffer
164 while len(buf) > 8:
165 if buf[0] != self.START:
166 buf.pop(0); continue
167 try:
168 l_val = struct.unpack("<H", buf[1:3])[0]
169 tot = 1 + 2 + l_val + 3
170 if len(buf) < tot: break
171
172 dest = struct.unpack("<H", buf[3:5])[0]
173 src = struct.unpack("<H", buf[5:7])[0]
174
175 # Filter addresses
176 if src not in [65535, MY_SA]: self.active_devices.add(src)
177 if dest not in [65535, MY_SA]: self.active_devices.add(dest)
178
179 del buf[:tot]
180 except: buf.pop(0)
181 except: pass
182
183 if not self.running: return
184
185 # Fallback mechanism
186 if not self.active_devices:
187 self.active_devices = {1, 100}
188 self._log("No traffic detected. Forcing scan on IDs 1 and 100.")
189 else:
190 self._log(f"Devices detected: {list(self.active_devices)}")
191
192 # --- PHASE 2: SCANNING ---
193 devs = sorted(list(self.active_devices))
194 total = len(devs)
195
196 for i, dev in enumerate(devs):
197 if not self.running: break
198 base_prog = 50 + (i / total * 50)
199 self._log(f"PHASE 2: Scanning Device {dev} ({i+1}/{total})", base_prog)
200
201 self._scan_smart(dev, 0, 1000)
202
203 self._log("SCAN COMPLETED!", 100)
204 self.running = False
205 if self.sock: self.sock.close()

207 def _scan_smart(self, target: int, start: int, end: int) -> None:
208 """
209 @brief Scans a memory range on a specific device.
210 @details Implements logic to skip large empty memory blocks to save time.
211 @param target The device address to scan.
212 @param start Start index.
213 @param end End index.
214 """
215 empty_streak = 0
216
217 for idx in range(start, end):
218 if not self.running: break
219
220 # Log every 10 items to avoid spamming the UI queue
221 if idx % 10 == 0:
222 self._log(f"Device {target} : Index {idx}...", None)
223
224 if empty_streak >= MAX_EMPTY_STREAK:
225 self._log(f"Device {target} : Empty zone detected. Skipping device.", None)
226 break
227
228 time.sleep(0.01) # Throttling
229
230 if self.sock:
231 # Flush buffer
232 self.sock.settimeout(0.01)
233 try:
234 while self.sock.recv(4096): pass
235 except: pass
236
237 # Build Request: [Start] [Len] [Dest] [Src] [Cmd=0x02] [Data] [CRC] [Stop]
238 req = struct.pack("<BH", 1, idx)
239 l = len(req) + 5
240 h = struct.pack("<HHHB", l, target, MY_SA, 0x02) + req
241 frame = struct.pack("B", self.START) + h + struct.pack(">H", self._crc(h)) + struct.pack("B", self.STOP)
242
243 found = False
244 try:
245 self.sock.send(frame)
246 self.sock.settimeout(0.2)
247 resp = self.sock.recv(1024)
248
249 # Parsing
250 if len(resp) > 8 and self.START in resp:
251 start_pos = resp.find(self.START)
252 # Ensure we have enough bytes for header
253 if start_pos >= 0 and len(resp) > start_pos + 8:
254 # Strict check: Function Code must be 0x82 (Read Response)
255 if resp[start_pos+7] == 0x82:
256 payload = resp[start_pos+8:-3]
257 res = self._decode(target, idx, payload)
258 if res:
259 self.result_queue.put(res)
260 found = True
261 empty_streak = 0
262 except: pass
263
264 if not found:
265 empty_streak += 1

267 def _decode(self, addr: int, idx: int, data: bytes) -> Optional[Dict[str, Any]]:
268 """
269 @brief Decodes a binary payload based on the protocol definition.
270 @param addr Device address.
271 @param idx Parameter index.
272 @param data Raw binary payload from the response.
273 @return Dictionary containing decoded fields (Name, Val, Exp, Unit, etc.) or None.
274 """
275 try:
276 cursor = 3
277 def read_str(buf: bytes, pos: int) -> Tuple[str, int]:
278 end = buf.find(b'\x00', pos)
279 if end == -1: return "", len(buf)
280 s = buf[pos:end].decode('latin-1', 'ignore').strip()
281 return s, end + 1
282
283 name, cursor = read_str(data, cursor)
284 unit, cursor = read_str(data, cursor)
285
286 if not name or name == "?": return None
287 if len(data) < cursor + 2: return None
288
289 info_raw = data[cursor]
290 exp_raw_byte = bytes([data[cursor+1]])
291 cursor += 2
292
293 type_id = info_raw & 0x0F
294 rw_bit = bool(info_raw & 0x20)
295
296 exponent = struct.unpack("<b", exp_raw_byte)[0]
297 if abs(exponent) > 6: exponent = 0
298
299 type_name, size, fmt, is_float = TYPE_DEFS.get(type_id, ("UNK", 0, None, False))
300
301 val_str = "---"
302 if size > 0 and len(data) >= cursor + size:
303 raw_bytes = data[cursor : cursor + size]
304 if fmt:
305 val = struct.unpack(fmt, raw_bytes)[0]
306
307 if is_float:
308 val_str = f"{val:.2f}"
309 elif type_id == 10:
310 val_str = "ON" if val else "OFF"
311 elif type_id == 12:
312 val_str = "TXT"
313 else:
314 if exponent != 0:
315 val_str = f"{val * (10 ** exponent):g}"
316 else:
317 val_str = f"{val}"
318
319 return {
320 "addr": addr, "idx": idx, "name": name,
321 "val": val_str, "exp": exponent, "unit": unit,
322 "type": type_name, "rw": "RW" if rw_bit else "RO"
323 }
324 except: return None
